8. Write a program to implement FCFS page replacement algorithm.

Program:
--------------------------------------------------

#include<stdio.h>
#include<conio.h>

int i,j,nof,nor,ref[50],frm[10],pf=0,victim=-1,flag;

void main()
{
    clrscr();

    printf("Enter no.of Frames:");
    scanf("%d",&nof);

    printf("Enter no.of Reference Strings:");
    scanf("%d",&nor);

    printf("Enter Reference String:");

    for(i=0;i<nor;i++)
        scanf("%d",&ref[i]);

    for(i=0;i<nof;i++)
        frm[i]=-1;

    printf("\nPage Replacement Process\n");

    for(i=0;i<nor;i++)
    {
        flag=0;

        for(j=0;j<nof;j++)
        {
            if(frm[j]==ref[i])
                flag=1;
        }

        if(flag==0)
        {
            victim=(victim+1)%nof;
            frm[victim]=ref[i];
            pf++;
        }

        for(j=0;j<nof;j++)
            printf("%4d",frm[j]);

        printf("\n");
    }

    printf("\nTotal Page Faults=%d",pf);

    getch();
}
