12. Write a program to implement Banker’s algorithm for deadlock avoidance.

Program:
--------------------------------------------------

#include<stdio.h>
#include<conio.h>

void main()
{
    int alloc[10][10],max[10][10],need[10][10];
    int avail[10],finish[10];
    int i,j,p,r,count=0,safe[10];

    clrscr();

    printf("Enter no.of Processes:");
    scanf("%d",&p);

    printf("Enter no.of Resources:");
    scanf("%d",&r);

    printf("\nEnter Allocation Matrix:\n");

    for(i=0;i<p;i++)
    {
        for(j=0;j<r;j++)
            scanf("%d",&alloc[i][j]);
    }

    printf("\nEnter Max Matrix:\n");

    for(i=0;i<p;i++)
    {
        for(j=0;j<r;j++)
            scanf("%d",&max[i][j]);
    }

    printf("\nEnter Available Resources:\n");

    for(i=0;i<r;i++)
        scanf("%d",&avail[i]);

    printf("\nNeed Matrix:\n");

    for(i=0;i<p;i++)
    {
        for(j=0;j<r;j++)
        {
            need[i][j]=max[i][j]-alloc[i][j];
            printf("%4d",need[i][j]);
        }

        printf("\n");
    }

    for(i=0;i<p;i++)
        finish[i]=0;

    while(count<p)
    {
        int found=0;

        for(i=0;i<p;i++)
        {
            if(finish[i]==0)
            {
                int flag=1;

                for(j=0;j<r;j++)
                {
                    if(need[i][j]>avail[j])
                    {
                        flag=0;
                        break;
                    }
                }

                if(flag==1)
                {
                    for(j=0;j<r;j++)
                        avail[j]+=alloc[i][j];

                    safe[count]=i;
                    count++;
                    finish[i]=1;
                    found=1;
                }
            }
        }

        if(found==0)
            break;
    }

    if(count==p)
    {
        printf("\nSystem is in Safe State\n");
        printf("Safe Sequence: ");

        for(i=0;i<p;i++)
            printf("P%d ",safe[i]);
    }
    else
    {
        printf("\nSystem is not in Safe State");
    }

    getch();
}
