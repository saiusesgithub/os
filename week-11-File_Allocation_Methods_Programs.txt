11.a) Write a C Program to implement Sequential File Allocation method.

Program:
--------------------------------------------------

#include<stdio.h>
#include<conio.h>

void main()
{
    int n,i,j,start[10],len[10];

    clrscr();

    printf("Enter no.of files:");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("\nEnter starting block and length of file %d:",i+1);
        scanf("%d%d",&start[i],&len[i]);
    }

    printf("\nFile\tStart\tLength\tBlocks\n");

    for(i=0;i<n;i++)
    {
        printf("%d\t%d\t%d\t",i+1,start[i],len[i]);

        for(j=0;j<len[i];j++)
            printf("%d ",start[i]+j);

        printf("\n");
    }

    getch();
}

11.b) AIM: Write a C Program to implement Indexed File Allocation method.

Program:
--------------------------------------------------

#include<stdio.h>
#include<conio.h>

void main()
{
    int n,i,j,index[10],len[10],block[10][10];

    clrscr();

    printf("Enter no.of files:");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("\nEnter index block and length of file %d:",i+1);
        scanf("%d%d",&index[i],&len[i]);

        printf("Enter blocks:");

        for(j=0;j<len[i];j++)
            scanf("%d",&block[i][j]);
    }

    printf("\nFile\tIndex\tBlocks\n");

    for(i=0;i<n;i++)
    {
        printf("%d\t%d\t",i+1,index[i]);

        for(j=0;j<len[i];j++)
            printf("%d ",block[i][j]);

        printf("\n");
    }

    getch();
}

11.C) AIM: Write a C Program to implement Linked File Allocation method.

Program:
--------------------------------------------------

#include<stdio.h>
#include<conio.h>

struct file
{
    int start,len,block[10];
}f[10];

void main()
{
    int n,i,j;

    clrscr();

    printf("Enter no.of files:");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("\nEnter start block and length of file %d:",i+1);
        scanf("%d%d",&f[i].start,&f[i].len);

        printf("Enter blocks:");

        for(j=0;j<f[i].len;j++)
            scanf("%d",&f[i].block[j]);
    }

    printf("\nFile\tStart\tLength\tBlocks\n");

    for(i=0;i<n;i++)
    {
        printf("%d\t%d\t%d\t",i+1,f[i].start,f[i].len);

        for(j=0;j<f[i].len;j++)
            printf("%d ",f[i].block[j]);

        printf("\n");
    }

    getch();
}
